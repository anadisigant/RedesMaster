QUESTÃO 18 – Estude os programas Python abaixo. Execute os programas e utilizando Wireshark  
apresente como acontece o estabelecimento da conexão, a troca de dados e o encerramento da  
conexão TCP
###########################################
# cliente_tcp_simples.py
import socket
HOST = 'localhost'
PORT = 5000
cliente = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
cliente.connect((HOST, PORT))
mensagem = "Olá, servidor!"
cliente.send(mensagem.encode())
cliente.close()
###########################################
# servidor_tcp_simples.py
import socket
HOST = 'localhost'
PORT = 5000
servidor = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
servidor.bind((HOST, PORT))
servidor.listen(1)
print("Aguardando conexão...")
conn, addr = servidor.accept()
print(f"Conectado por {addr}")
dados = conn.recv(1024)
print("Mensagem recebida:", dados.decode())
conn.close()
servidor.close()
